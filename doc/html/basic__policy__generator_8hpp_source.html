285void BasicPolicyGenerator< CommandSystem >::setStates(const std::vector< float >& states, const std::vector< float >& velocities, const std::vector< float >& accelerations) {
286 if (states.size() != size()) {
287 throw std::invalid_argument("Invalid command vector dimension!");
288 }
289 for (size_t idx = 0; idx < size(); idx++) {
290 _command_systems[idx]->value = states[idx];
291 }
292
293 if (velocities.size() != size()) {
294 throw std::invalid_argument("Invalid velocity vector dimension!");
295 }
296 for (size_t idx = 0; idx < size(); idx++) {
297 _command_systems[idx]->dvalue = velocities[idx];
298 }
299
300 if (accelerations.size() != size()) {
301 throw std::invalid_argument("Invalid acceleration vector dimension!");
302 }
303 for (size_t idx = 0; idx < size(); idx++) {
304 _command_systems[idx]->ddvalue = accelerations[idx];
305 }
306}
